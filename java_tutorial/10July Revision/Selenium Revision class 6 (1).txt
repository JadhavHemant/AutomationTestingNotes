How to handle Multiple Windows or Tabs in Selenium
----------------------------------------------------------------------------------------------------------------------------------------------
	- we handle multiple window in selenium by using getWindowHandle() and getWindowHandles() methods.
	- getWindowHandle() is used to capture the current window id number which is combination of int and character.
	- getWindowHandles() methods is used to capture  the window id number of all window.
	- once we capture any window id number then we switch focus to window using switchTo().window() by passing window id number.
	-return type of getWindowHandle() methods is String.
	- return type of getWindowHandles() methods is Set<String>.
	- Set Interface implements classes does not store duplicate values, as well as it not insert the values as per the index position.
	- so once the capture all window id number then we convert it into the ArrayList object.
	- why we convert to the ArrayList object because the ArrayList class , it insert the values as per the index position, as well as we capture the ArrayList object values using get() as per the index position, in Set implemented classes there is no get() method to capture the values from Set implemented classes, so we convert Set<String> into the ArrayList<String>.

package Tutorial6;

import java.util.ArrayList;
import java.util.Set;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;

public class Demo1 {

	public static void main(String[] args) throws InterruptedException {
		// Step 1: up casting
		WebDriver driver = new ChromeDriver();

		// step 2: open a url
		driver.get("https://prafpawar11.github.io/multiplewindows.html");

		// step 3: find and enter value in first name text box
		driver.findElement(By.name("fname")).sendKeys("Pooja");

		Thread.sleep(5000);
		// step 4: find and click on orange HRM Link
		driver.findElement(By.partialLinkText("Orange")).click();

		// step 5: capture all window id number
		Set<String> allwindow = driver.getWindowHandles();

		// step 6: convert the Set<String> into ArrayList<String>
		ArrayList<String> allArrayWindow = new ArrayList<String>(allwindow);

		// step 7: capture orange HRM Window Id number
		String orangeWindowID = allArrayWindow.get(1);

		// step 8: switch focus to orange hrm window
		driver.switchTo().window(orangeWindowID);
		Thread.sleep(5000);
		// step 9: find and enter valid credentails
		driver.findElement(By.name("username")).sendKeys("Admin");
		driver.findElement(By.name("password")).sendKeys("admin123");
		Thread.sleep(3000);
		driver.findElement(By.xpath("//button[text()=' Login ']")).click();
		Thread.sleep(3000);
		// step 10: capture main window ID
		String mainWindowID = allArrayWindow.get(0);

		// step 11: switch focus to main window
		driver.switchTo().window(mainWindowID);

		// step 12: find and enter value in last name text box
		driver.findElement(By.name("lname")).sendKeys("Kale");

	}

}
