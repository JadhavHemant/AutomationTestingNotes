package assignments;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;

public class CalendarAssignment {

	public static void main(String[] args) throws InterruptedException {

		// connect to browser

		// up casting
		WebDriver driver = new ChromeDriver();

		// open url
		driver.get("https://www.goibibo.com/");

		Thread.sleep(7000);

		// find date webElement
		WebElement wb = driver.findElement(By
				.xpath("//span[text()='Departure']/following-sibling::p[@class='sc-12foipm-4 czGBLf fswWidgetTitle']"));

		wb.click();

		// run infinite while loop
		while (true) {
			// find and capture month and year value
			WebElement wb1 = driver.findElement(By.xpath(
					"//div[@class='DayPicker-Month']/preceding-sibling::div/child::div[@class='DayPicker-Caption']/div"));
			String monthYear = wb1.getText();
			System.out.println(monthYear);

			if (monthYear.equals("January 2025")) {
				WebElement wb2 = driver.findElement(
						By.xpath("//div[@class='DayPicker-Month']/preceding-sibling::div/descendant::p[text()='30']"));

				wb2.click();

				break;
			} else {
				// find the next button and click on it
				WebElement wbNext = driver
						.findElement(By.xpath("//span[@class='DayPicker-NavButton DayPicker-NavButton--next']"));
				wbNext.click();
			}

		}
	}

}
