package javaScriptExecutor;

import org.openqa.selenium.By;
import org.openqa.selenium.JavascriptExecutor;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;

public class ScrollIntoView {

	public static void main(String[] args) {

		//connect to browser
		
		//up casting
		WebDriver driver = new ChromeDriver();
		
		//open url
		driver.get("https://www.amazon.in");
		
		//find webelement in web page
		WebElement wb = driver.findElement(By.xpath("//span[text()='Best Sellers in Clothing & Accessories']"));
		
		//convert webdriver into JavascriptExecutor
		JavascriptExecutor js = (JavascriptExecutor)driver;
		
		//use executeScript() by passing arguments[0].scrollIntoView and webElement instance
		js.executeScript("arguments[0].scrollIntoView();", wb);
	}

}
