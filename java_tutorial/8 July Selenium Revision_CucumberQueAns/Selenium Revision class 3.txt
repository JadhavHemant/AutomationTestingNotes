How to handle drop down in selenium?
		or
how to select values from drop down?
--------------------------------------------------------------------------------------------------------------------
	- first we find drop down by using findElement() method and using given locator

	- there are 6 ways we can select values from drop down if drop down HTML DOM structure element start with select tag in html.

	1) selectByVisibleText() method
	2) selectByValue() method
	3) selectByIndex() method
	4) by iterate all drop down values then by using condition
	5) by using Actions class methods.
	6) by using JavascriptExecutor, executeScript() method.
---------------------------------------------------------- 
	if drop down tag starts with select tag in HTML DOM structure then we select the drop down using select class from selenium.	

1) selectByVisibleText() method
	- first we find drop down by using findElement() method and using given locator
	- then we have to create object of Select class by passing Drop Down WebElement instance name.
	- then we use selectByVisibleText() method by passing physical drop down values.

	WebElement wb = driver.findElement(By.name(""));
	Select sel =new Select(wb);
	sel.selectByVisibleText("Mar");

	- find month drop down and select value.

		WebElement wb = driver.findElement(By.name(""));

		Select sel =new Select(wb);

step 5: select values using selectByVisibleText() method by passing drop down physical values

		sel.selectByVisibleText("Mar");

	}

package Tutorial3;

import java.util.List;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;

public class Demo2 {

	public static void main(String[] args) throws InterruptedException {
		// Step 1: up casting
		WebDriver driver = new ChromeDriver();

		// step 2: open a url
		driver.get("https://www.facebook.com/reg");

		// step 3: find the drop down element using findElement() and given locator
		WebElement wb = driver.findElement(By.name("birthday_month"));

		// step 4: click on drop down
		wb.click();

		Thread.sleep(4000);
		// step 5: find all drop down values
		List<WebElement> listDropDown = driver.findElements(By.xpath("//select[@name='birthday_month']/child::option"));

		// step 6: iterate all drop down values
		for (WebElement dropdown : listDropDown) {
			// step 7: capture the actual drop down value.
			String actualValue = dropdown.getText();

			// step 8: compare actual value with expected value
			if (actualValue.equalsIgnoreCase("Dec")) {
				// step 9: click on drop down value
				dropdown.click();

				// step 10: break the loop
				break;
			}
		}

	}

}

	- then use keyDown() and by passing Keys arrow up or down and then use Keys.Enter and build and perform method

package Tutorial3;

import org.openqa.selenium.By;
import org.openqa.selenium.Keys;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.interactions.Actions;

public class Demo3 {

	public static void main(String[] args) throws InterruptedException {

		// Step 1: up casting
		WebDriver driver = new ChromeDriver();

		// step 2: open a url
		driver.get("https://www.facebook.com/reg");

		// step 3: find the drop down element using findElement() and given locator
		WebElement wb = driver.findElement(By.name("birthday_month"));

		// step 4: create object of Actions class by passing WebDriver instance
		Actions act = new Actions(driver);

		// step 5: click on drop down using click() method from Actions class or
		// WebElement
		act.click(wb).build().perform();
		Thread.sleep(3000);
		// step 6: select April month value from drop down using keyDown by passing
		// Keys.ARROW_UP

		act.keyDown(Keys.ARROW_UP).build().perform();
		Thread.sleep(1000);

		act.keyDown(Keys.ARROW_UP).build().perform();
		Thread.sleep(1000);

		act.keyDown(Keys.ARROW_UP).build().perform();
		Thread.sleep(1000);

		act.sendKeys(Keys.ENTER).build().perform();
	}
}

package Tutorial3;

import java.time.Duration;
import java.util.List;

import org.openqa.selenium.By;
import org.openqa.selenium.Keys;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;

public class Demo2 {

	public static void main(String[] args) throws InterruptedException {
		// Step 1: up casting
		WebDriver driver = new ChromeDriver();
		driver.manage().timeouts().implicitlyWait(Duration.ofSeconds(30)).pageLoadTimeout(Duration.ofSeconds(30));
		// step 2: open a url
		driver.get("https://www.vsp.com/create-account");
		Thread.sleep(7000);

		// step 3: find the drop down element using findElement() and given locator
		WebElement wb = driver.findElement(By.name("preferredLanguage"));

		// step 4: click on drop down
		wb.click();

		Thread.sleep(4000);
		// step 5: find all drop down values
		List<WebElement> listDropDown = driver
				.findElements(By.xpath("//span[@class='ng-option-label ng-star-inserted']"));

		// step 6: iterate all drop down values
		for (WebElement dropdown : listDropDown) {
			// step 7: capture the actual drop down value.
			String actualValue = dropdown.getText();

			// step 8: compare actual value with expected value
			if (actualValue.equalsIgnoreCase("Farsi")) {
				// step 9: click on drop down value
				dropdown.click();

				// step 10: break the loop
				break;
			}
		}

	}

}
